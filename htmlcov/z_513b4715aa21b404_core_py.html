30def generate_high_quality_prompt(user_input: str, options: Dict[str, Any], mode: str = 'basic', persona: Optional[Dict[str, Any]] = None) -> Dict[str, Any]: 

31 """ 

32 사용자의 요구사항과 옵션을 바탕으로 고퀄리티 프롬프트 초안을 생성한다.

48**LLM 챗봇에게 입력할 수 있는, 고도화되고 구조화된 프롬프트 문장(질문/지시문)을** 만들어주는 것입니다.

80 사용자 요청: 

81 {user_input} 

82  

83 선택된 옵션:

194# QA 평가자 역할: 품질 평가/점수화/개선점 제안

196 """ 

197 프롬프트의 명확성, 구체성, LLM 적합성 등 품질 평가 및 개선점 제안 

198 Returns: { 'score': int, 'review': str, 'improvement': str } 

199 """ 

200 client = get_openai_client() 

201 if not client: 

202 return { 

203 'score': 80, 

204 'review': "OpenAI 연동 미설정으로 인한 기본 평가", 

205 'improvement': "OpenAI API 키 설정 후 재평가 필요" 

206 } 

207 

208 try: 

209 system_prompt = """당신은 프롬프트 품질 평가 전문가입니다.  

210 주어진 프롬프트의 품질을 종합적으로 평가해주세요. 

211  

212 평가 기준 (각 항목 20점 만점): 

213 1. 명확성: 목적과 요구사항이 명확한가? 

214 2. 구체성: 구체적인 지시사항과 예시가 포함되어 있는가? 

215 3. 구조화: 논리적으로 구조화되어 있는가? 

216 4. 맥락 제공: 필요한 배경 정보가 충분한가? 

217 5. 출력 형식: 원하는 출력 형식이 명시되어 있는가? 

218  

219 총점: 100점 만점 

220 등급: 90-100점(A), 80-89점(B), 70-79점(C), 60-69점(D), 60점 미만(F) 

221  

222 응답 형식: 

223 - 점수: 숫자 (0-100) 

224 - 평가: 상세한 평가 내용 

225 - 개선점: 구체적인 개선 제안""" 

226 

227 user_message = f""" 

228 평가할 프롬프트: 

229 {prompt} 

230  

231 위 프롬프트를 품질 평가 기준에 따라 평가해주세요. 

232 """

234 response = client.chat.completions.create( 

235 model="gpt-4", 

236 messages=[ 

237 {"role": "system", "content": build_personalized_prompt(system_prompt, persona)}, 

238 {"role": "user", "content": user_message} 

239 ], 

240 max_tokens=1000,

242 ) 

243

279# 프롬프트 저장/로드 기능 

280def save_prompt_to_file(prompt_data: Dict[str, Any], filename: str = None) -> str: 

281 """프롬프트를 파일로 저장""" 

282 if not filename: 

283 import datetime 

284 timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S") 

285 filename = f"prompt_{timestamp}.json" 

286 

287 # prompts 디렉토리 생성 

288 prompts_dir = "prompts" 

289 os.makedirs(prompts_dir, exist_ok=True) 

290 

291 filepath = os.path.join(prompts_dir, filename) 

292 

293 try: 

294 with open(filepath, 'w', encoding='utf-8') as f: 

295 json.dump(prompt_data, f, ensure_ascii=False, indent=2) 

296 return filepath 

297 except Exception as e: 

298 logger.error(f"프롬프트 저장 중 오류: {e}") 

299 return ""

302 """파일에서 프롬프트 로드"""
