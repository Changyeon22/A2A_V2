26def summarize_text(text_to_summarize: str, prompt_template: str = DEFAULT_PROMPT_TEMPLATE) -> Dict[str, Any]: 

27 """ 

28 주어진 텍스트를 OpenAI API를 사용하여 요약합니다. 

29 재시도 로직이 포함되어 있어 API 속도 제한(rate limit) 오류 발생 시 자동으로 재시도합니다. 

30 

31 Args: 

32 text_to_summarize (str): 요약할 텍스트 내용. 

33 prompt_template (str): 요약 프롬프트의 템플릿. 

34 

35 Returns: 

36 dict: 상태와 요약 결과 또는 오류 메시지를 포함하는 딕셔너리. 

37 """ 

38 logger.info(LOG_START) 

39 

40 if not text_to_summarize or not text_to_summarize.strip(): 

41 logger.warning(LOG_EMPTY_TEXT) 

42 return {"status": STATUS_ERROR, "message": ERROR_EMPTY_TEXT} 

43 

44 # 재시도 설정 

45 max_retries = 5 

46 base_delay = 1 # 초 단위 기본 대기 시간 

47 

48 # 텍스트 준비 및 프롬프트 생성 (재시도와 무관하게 한 번만 수행) 

49 truncated_text = prepare_text_for_summarization(text_to_summarize) 

50 full_prompt = create_summary_prompt(truncated_text, prompt_template) 

51 

52 for attempt in range(max_retries): 

53 try: 

54 # OpenAI 클라이언트 초기화 - 함수 내부에서 호출할 때마다 생성 

55 # 환경 변수에서 API 키를 가져오도록 해서 보안성 강화 

56 try: 

57 import os 

58 api_key = os.environ.get("OPENAI_API_KEY") 

59 if not api_key: 

60 logger.error("OpenAI API 키가 환경변수에 없습니다") 

61 return {"status": STATUS_ERROR, "message": "API 키 구성 오류"} 

62 

63 client = OpenAI(api_key=api_key) 

64 except Exception as setup_err: 

65 logger.error(f"OpenAI 클라이언트 초기화 오류: {setup_err}") 

66 return {"status": STATUS_ERROR, "message": f"API 클라이언트 초기화 오류: {setup_err}"} 

67 

68 # OpenAI API 호출 

69 logger.info(f"{LOG_API_CALL} (시도 {attempt + 1}/{max_retries})") 

70 response = client.chat.completions.create( 

71 model=DEFAULT_MODEL, 

72 messages=[ 

73 {"role": "system", "content": SYSTEM_MESSAGE}, 

74 {"role": "user", "content": full_prompt} 

75 ], 

76 max_tokens=DEFAULT_MAX_TOKENS, 

77 temperature=DEFAULT_TEMPERATURE, 

78 ) 

79 

80 summary = response.choices[0].message.content.strip() 

81 logger.info(LOG_COMPLETE) 

82 return {"status": STATUS_SUCCESS, "summary": summary} 

83 

84 except Exception as e: 

85 # 에러 유형 검사 

86 error_str = str(e).lower() 

87 

88 # 속도 제한(Rate Limit) 관련 오류 

89 if 'rate limit' in error_str or 'rate_limit' in error_str or 'too many requests' in error_str: 

90 wait_time = base_delay * (2 ** attempt) # 지수 백오프 

91 error_msg = f"속도 제한(Rate Limit) 오류 발생: {e}. {wait_time}초 후 재시도합니다. (시도 {attempt + 1}/{max_retries})" 

92 logger.warning(error_msg) 

93 

94 if attempt < max_retries - 1: # 마지막 시도가 아니면 대기 후 재시도 

95 time.sleep(wait_time) 

96 continue 

97 else: # 마지막 시도면 실패 반환 

98 return { 

99 "status": STATUS_ERROR, 

100 "message": f"API 속도 제한(Rate Limit) 오류: {max_retries}회 시도 후 실패했습니다. 잠시 후 다시 시도해주세요.", 

101 "retry_after": 60 # 사용자에게 60초 후에 다시 시도하라고 제안 

102 } 

103 

104 # API 오류 처리 - 서버 오류 

105 elif any(x in error_str for x in ['500', '502', '503', '504', 'server error', 'timeout']): 

106 error_msg = LOG_API_ERROR.format(e) 

107 logger.error(error_msg) 

108 logger.debug(traceback.format_exc()) 

109 

110 if attempt < max_retries - 1: # 마지막 시도가 아니면 대기 후 재시도 

111 wait_time = base_delay * (2 ** attempt) 

112 logger.info(f"API 서버 오류, {wait_time}초 후 재시도합니다. (시도 {attempt + 1}/{max_retries})") 

113 time.sleep(wait_time) 

114 continue 

115 else: # 마지막 시도면 실패 반환 

116 return {"status": STATUS_ERROR, "message": ERROR_API.format(e)} 

117 

118 # 기타 API 오류 

119 else: 

120 error_msg = LOG_API_ERROR.format(e) 

121 logger.error(error_msg) 

122 logger.debug(traceback.format_exc()) 

123 return {"status": STATUS_ERROR, "message": ERROR_API.format(e)}

177 for name in TOOL_MAP:
